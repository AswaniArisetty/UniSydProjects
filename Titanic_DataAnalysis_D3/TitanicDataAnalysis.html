def age_groups(age):
    if age<=15:
        return 0
    elif age>15 and age <=40:
        return 1
    elif age >40 and age<=60:
        return 2
    elif np.isnan(age):
        return None
    else:
        return 3
    return none 

def gender_map(sex):
    if sex=='female':
        return 0
    elif sex=='male':

def survivors_by_group(group_by_field,survived):
    new_df=titanic_data.groupby(group_by_field)
    return_data={}
    #print "==============================="
    #print "Grouping by:",group_by_field," and filtering on survived:",survived
    for groups in new_df.groups.keys():
        group_data=titanic_data.groupby(group_by_field).get_group(groups)
        survival_group_data=group_data[group_data["Survived"]==survived]
        return_data[groups]=survival_group_data

    return return_data

Let us perform a hypothetical test with Null and Alternate hypotheses as below

Let Px denote the survival probability of a female passenger Let Py denote the survival probability of a male passenger

Null hypotheses H0: Px-Py <= 0 Alternate Hypotheses HA: Px-Py > 0

Below is a statistical test for difference between two proportions (Reference : Statistics for Engineers and Scientists by William Navidi 3rd edition)

print Px , Py

# Calculate the pooled proportion 
Pooled_proportion=(Total_male_survivors+Total_female_survivors)/(Total_male_passengers+Total_female_passengers)
print Pooled_proportion

#The value of standard deviation
#variance=(Pooled_proportion*(1-Pooled_proportion))*(1/Total_male_passengers)+(1/Total_female_passengers)
Std_dev=sqrt((Pooled_proportion*(1-Pooled_proportion))*(1/Total_male_passengers)+(1/Total_female_passengers))
print Std_dev 

#Mean value of Null distribution for Px-Py is 0 and for the observed is 
z=((Px-Py)-0)/Std_dev
